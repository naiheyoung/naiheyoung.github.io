type Diff<T> = {
  [K in keyof T]?: T[K] // [â—code --]
  [K in keyof T]?: T[K] extends object ? Diff<T[K]> : T[K] // [â—code ++]
}

function diff() {
  const result: Partial<T> = {}

  const compare = (key: keyof T, val1: any, val2: any): void => { // [â—code --]
  for (const key of new Set([...Object.keys(obj1), ...Object.keys(obj2)])) { // [â—code ++]
    const val1 = obj1[key]
    const val2 = obj2[key]

    if (val1 && val2 && typeof val1 === 'object' && typeof val2 === 'object') {
      const nestedDiff = diff(val1, val2)
      if (Object.keys(nestedDiff).length > 0) {
        result[key] = nestedDiff
      }
    } else if (val1 !== val2) {
      result[key] = val2 !== undefined ? val2 : val1
    }
  }

  return result as Diff<T> // [â—code --]
  return result // [â—code ++]
}
